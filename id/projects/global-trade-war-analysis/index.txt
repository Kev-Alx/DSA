```python
class TradeWarDataLoader:
    """
    A robust, scalable data ingestion pipeline for macroeconomic and financial datasets.
    Handles timezone normalization, missing value imputation, and memory optimization.
    """
    def __init__(self, data_dir='data/'):
        self.data_dir = data_dir
        self.data = {}

        self.file_map = {
            'tariff_timeline': 'tariff_timeline.csv',
            'stock_market': 'stock_market_reaction.csv',
            'currency': 'currency_impact.csv',
            'sector': 'sector_impact.csv',
            'inflation': 'inflation_response.csv',
            'trade_volume': 'trade_volume_annual.csv'
        }

    def ingest_and_clean(self):
        """Loads CSVs, parses dates, and enforces data types."""
        print("🔄 Initiating Data Ingestion Pipeline...")

        for name, filename in self.file_map.items():
            try:
                filepath = f"{self.data_dir}{filename}"
                df = pd.read_csv(filepath)

                if 'date' in df.columns:
                    df['date'] = pd.to_datetime(df['date'], errors='coerce')
                    df = df.sort_values('date').reset_index(drop=True)

                self.data[name] = df

            except FileNotFoundError:
                print(f"⚠️ Warning: {filename} not found in {self.data_dir}.")
            except Exception as e:
                print(f"⚠️ Error loading {filename}: {str(e)}")

        print("✅ Data Ingestion Complete.")

    def get_portfolio_summary(self):
        """Generates an executive summary of the loaded data architecture."""
        print("📊 DATA ARCHITECTURE SUMMARY")
        print("="*60)

        summary_data = []
        for name, df in self.data.items():
            date_range = "N/A"
            if 'date' in df.columns and not df['date'].empty:
                start_dt = df['date'].min().strftime('%Y-%m-%d')
                end_dt = df['date'].max().strftime('%Y-%m-%d')
                date_range = f"{start_dt} to {end_dt}"

            summary_data.append({
                'Dataset': name.replace('_', ' ').title(),
                'Records': f"{len(df):,}",
                'Features': len(df.columns),
                'Date Coverage': date_range
            })

        summary_df = pd.DataFrame(summary_data)
        print(summary_df.to_string(index=False, justify='left'))

    def get_dataset(self, name):
        """Safe getter for datasets."""
        return self.data.get(name, pd.DataFrame())
```

```python
def perform_audit(df_dict):
    """
    Audits financial datasets for missing trading days, NaN values,
    and extreme outliers that could break GARCH/VAR models.
    """
    for name, df in df_dict.items():
        if df.empty:
            continue

        missing_pct = df.isnull().mean().max() * 100
        if missing_pct > 5:
            print(f"  [!] {name}: High missing data detected (Max {missing_pct:.1f}% in a single column). Imputation required.")

        num_cols = df.select_dtypes(include=[np.number]).columns
        if np.isinf(df[num_cols]).values.any():
            print(f"  [!] {name}: Infinite values detected. Models will crash. Fixing...")
            df.replace([np.inf, -np.inf], np.nan, inplace=True)

        if name in ['stock_market', 'currency'] and 'date' in df.columns:
            initial_len = len(df)
            df = df[df['date'].dt.dayofweek < 5]
            dropped = initial_len - len(df)
            if dropped > 0:
                print(f"  [*] {name}: Removed {dropped} weekend observations to align trading days.")

perform_audit({
    'Equities': df_stocks,
    'FX': df_fx,
    'Macro': df_macro,
    'Tariffs': df_tariffs
})

if not df_tariffs.empty:
    major_shocks = df_tariffs[df_tariffs['tariff_rate_delta'] >= 10.0].copy()
    event_dates = major_shocks['date'].sort_values().unique()
    print(f"📌 Flagged {len(event_dates)} unique Major Tariff Shock dates for the Event Study.")
```

```python
def calculate_caar(df_stocks, index_name, event_dates, est_window=90, evt_window=5):
    """
    Calculates Cumulative Average Abnormal Returns (CAAR) for a given index.
    Uses a Mean-Adjusted Return model based on the estimation window.
    """
    df_idx = df_stocks[df_stocks['index_name'] == index_name].copy()
    df_idx = df_idx.set_index('date').sort_index()
    df_idx['daily_return_pct'] = df_idx['daily_return_pct'].fillna(0)

    event_cars = []

    for event_date in event_dates:
        event_date = pd.to_datetime(event_date)

        # Safely get integer locations for modern Pandas versions
        try:
            event_loc = df_idx.index.get_indexer([event_date], method='nearest')[0]
            if event_loc == -1:
                continue
        except KeyError:
            continue

        est_start = max(0, event_loc - evt_window - est_window)
        est_end = max(0, event_loc - evt_window)

        evt_start = max(0, event_loc - evt_window)
        evt_end = min(len(df_idx) - 1, event_loc + evt_window)

        # Estimation window to calculate baseline expected return
        estimation_data = df_idx.iloc[est_start:est_end]
        if len(estimation_data) < (est_window // 2):
            continue

        expected_return = estimation_data['daily_return_pct'].mean()

        # Event window calculations
        event_data = df_idx.iloc[evt_start:evt_end + 1].copy()
        if len(event_data) != (2 * evt_window + 1):
            continue

        event_data['abnormal_return'] = event_data['daily_return_pct'] - expected_return
        event_data['CAR'] = event_data['abnormal_return'].cumsum()
        event_data['event_day'] = range(-evt_window, evt_window + 1)

        event_cars.append(event_data.set_index('event_day')['CAR'])

    if not event_cars:
        return None

    # Aggregate to CAAR
    car_df = pd.concat(event_cars, axis=1)
    caar = pd.DataFrame({
        'CAAR': car_df.mean(axis=1),
        'STD': car_df.std(axis=1),
        'N': car_df.count(axis=1)
    })

    # 95% Confidence Interval
    t_crit = 1.96
    caar['CI_upper'] = caar['CAAR'] + (t_crit * (caar['STD'] / np.sqrt(caar['N'])))
    caar['CI_lower'] = caar['CAAR'] - (t_crit * (caar['STD'] / np.sqrt(caar['N'])))

    return caar

# Execute CAAR calculation for S&P 500
caar_sp500 = calculate_caar(df_stocks, 'S&P 500 (USA)', event_dates, est_window=90, evt_window=5)

# Plotting the Event Study
if caar_sp500 is not None:
    fig, ax = plt.subplots()
    ax.plot(caar_sp500.index, caar_sp500['CAAR'], color='#002E5D', lw=2.5, label='CAAR')
    ax.fill_between(caar_sp500.index, caar_sp500['CI_lower'], caar_sp500['CI_upper'],
                    color='#6A737B', alpha=0.3, label='95% Confidence Interval')

    ax.axvline(0, color='#A2142F', linestyle='--', lw=2, label='Tariff Announcement (t=0)')
    ax.axhline(0, color='black', lw=1)

    ax.set_title("S&P 500 Cumulative Average Abnormal Returns (CAAR)\nAround Major Tariff Announcements")
    ax.set_xlabel("Trading Days Relative to Announcement")
    ax.set_ylabel("Cumulative Abnormal Return (%)")
    ax.legend(loc='upper left')
    plt.tight_layout()
    plt.show()
```

To evaluate the immediate equity market pricing of tariff implementations, a Cumulative Average Abnormal Return (CAAR) methodology is utilized. CAAR aggregates the daily impact of a specific event on asset prices by extracting expected benchmark returns, thereby isolating the abnormal pricing reaction directly attributable to the catalyst.

The wide 95% confidence band completely encompassing the zero line throughout the entire event window from the chart indicates that major tariff announcements do not produce a statistically significant directional shock to the aggregate S&P 500.

Key Findings:

1. Markets don’t really react sharply to tariff announcements at the index level. Big indices like the S&P 500 tend to absorb or already price in the news, so you don’t see a clear jump or drop in the few days around the event.
2. What does change is the level of uncertainty. After tariffs are announced, market movements become more spread out, suggesting increased volatility as money shifts between sectors that are hurt and those that benefit.

##### **What is GARCH?**

To quantify the transmission of trade policy uncertainty into foreign exchange markets, a Generalized Autoregressive Conditional Heteroskedasticity (GARCH) model is applied to the USD/CNY exchange rate returns. GARCH models time-varying volatility, specifically capturing the phenomenon of "variance clustering"—where large price fluctuations are followed by large fluctuations, and small by small. By estimating conditional variance, this methodology identifies how long a shock to the currency market persists before reverting to the mean.

##### **Analysis**

The model yields a highly significant $\beta$ coefficient of 0.8929 and an $\alpha$ coefficient of 0.0927. The sum of these parameters ($\alpha + \beta \approx 0.985$) approaches unity, indicating extreme volatility persistence. A tariff-induced shock to the USD/CNY exchange rate does not dissipate quickly; rather, it fundamentally alters the risk regime for an extended duration. This is visually confirmed by the sustained volatility clusters aligning directly with major tariff announcements across the historical timeline.

Furthermore, the post-tariff depreciation matrix and the primary exchange rate trajectory reveal a distinct structural break in currency market behavior. During the initial phases of the trade war (2018–2024), targeted tariffs consistently triggered CNY depreciation (ranging from 0.04% to 2.95% over 30 days), acting as a natural pricing shock absorber for exporters. However, the implementation of broad "All Goods" tariffs beginning in early 2025 marks a definitive regime shift, paradoxically resulting in immediate CNY appreciation (-0.23% to -1.55%) following the announcements.

#### **Section Summary**

1.  **High Volatility Persistence:** GARCH(1,1) estimations confirm that trade-induced FX volatility is highly persistent. Sudden geopolitical escalations create prolonged periods of currency instability rather than transient, quickly resolved pricing anomalies.
2.  **Currency as an Absorption Mechanism:** Early, targeted tariffs (2018–2024) systematically precipitated predictable CNY depreciation, effectively neutralizing a portion of the nominal tariff impact and shifting the cost burden dynamically.
3.  **Structural Regime Shift:** The transition to sweeping "All Goods" tariffs in 2025 triggered a definitive break in foreign exchange mechanics. The resulting contrarian CNY appreciation signals a fundamental exhaustion of the currency's capacity to unilaterally absorb broad-based macroeconomic trade barriers.

#### Causal Impact Evaluation using Difference-in-Differences (DiD) Modeling

Difference-in-Differences (DiD) is a methodology utilized to isolate and estimate the causal effect of a specific intervention. In this context, the implementation of a tariff. The model compares the post-event trajectory of a "treatment" group (sectors highly exposed to tariffs) against a "control" group (domestically insulated sectors). By evaluating the differential in performance before and after the event, the DiD estimator strips out broader market movements that affect all sectors simultaneously, revealing the isolated impact of the policy.

The Ordinary Least Squares (OLS) estimation yields an interaction coefficient—the core DiD estimator—of 0.0341. However, with a p-value of 0.831, this result is highly statistically insignificant. This indicates that the initial tariff implementation did not exert a definitive, causal drag on the returns of targeted sectors relative to insulated sectors in the immediate aftermath.

---

#### Section Summary

1. When a new tariff is introduced, the immediate impact on the targeted sector is not as severe as many would expect. There is no clear evidence of a sharp, direct drop in performance compared with sectors that are less exposed.

2. Over longer periods of trade tension, the picture changes. Sectors that are most exposed to tariffs—such as semiconductors and automobiles—have actually delivered stronger returns than more insulated industries. Investors seem willing to accept the extra uncertainty because those sectors offer greater upside.

3. The bigger effect of trade conflict shows up in volatility rather than broad market weakness. Trade friction tends to create larger swings in supply-chain-heavy industries, while capital often shifts toward higher-risk, higher-growth sectors instead of more defensive domestic names.

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller
from statsmodels.tsa.vector_ar.var_model import VAR

def check_stationarity(series, signif=0.05):
    """
    Returns True if the series is stationary at the given significance level.
    Uses the Augmented Dickey-Fuller test with automatic lag selection (AIC).
    """
    result = adfuller(series.dropna(), autolag='AIC')
    return result[1] <= signif


def find_date_col(df):
    """
    Returns the name of a parseable date column, or None if the index is
    already a DatetimeIndex. Uses a strict allowlist to avoid accidentally
    matching categorical columns like 'period' or 'era'.
    """
    if isinstance(df.index, pd.DatetimeIndex):
        return None
    for col in df.columns:
        if col.lower() in ('date', 'time', 'datetime', 'timestamp'):
            return col
    return None



def prepare_macro_data(df_macro, df_tariffs, df_trade):
    """
    Merges three disparate macroeconomic datasets into a unified monthly
    time-series suitable for Vector Autoregression (VAR) modelling.

    Handles:
      - df_macro  : long-format CPI/PPI panel with a 'date' column
      - df_tariffs: event-level tariff log with a 'date' column
      - df_trade  : annual trade data keyed by integer 'year' (no date column)
    """
    print("🔄 Standardizing Macro Time-Series for VAR...")

    df_t = df_tariffs.copy()
    df_t['date'] = pd.to_datetime(df_t['date'])
    df_t = df_t.set_index('date')

    monthly_tariffs = (
        df_t.resample('ME')['tariff_rate_pct']
        .mean()
        .fillna(0)
        .cumsum()
    )
    monthly_tariffs.name = 'Cumulative_Tariff_Rate'

    df_m = df_macro.copy()
    df_m['date'] = pd.to_datetime(df_m['date'])
    df_m = df_m.set_index('date')

    if 'metric' in df_m.columns:
        df_cpi = df_m[df_m['metric'].str.upper() == 'CPI'].copy()
        if df_cpi.empty:
            first_metric = df_m['metric'].iloc[0]
            print(f"  ⚠️  'CPI' not found in metric column — using '{first_metric}' instead.")
            df_cpi = df_m[df_m['metric'] == first_metric].copy()
    else:
        df_cpi = df_m.copy()

    monthly_cpi = (
        df_cpi.resample('ME')['value']
        .mean()
        .ffill()
    )
    monthly_cpi.name = 'CPI_Inflation'

    df_tr = df_trade.copy()
    df_tr['date'] = pd.to_datetime(df_tr['year'].astype(str) + '-01-01')
    df_tr = df_tr.set_index('date')

    monthly_trade = (
        df_tr.resample('ME')['value']
        .mean()
        .interpolate(method='linear')
    )
    monthly_trade.name = 'Global_Trade_Volume'

    df_var = pd.concat([monthly_tariffs, monthly_cpi, monthly_trade], axis=1).dropna()

    if df_var.empty:
        print("  ⚠️  No overlapping date range across the three datasets.")
    else:
        print(f"  ✅ VAR dataset ready | Shape: {df_var.shape} | "
              f"Range: {df_var.index.min().date()} → {df_var.index.max().date()}")

    return df_var



def _get_irf_bands(irf_obj, impulse_idx, response_idx, orth=True, signif=0.05):
    """
    Extracts point estimates and confidence bands for one impulse→response pair.
    Uses asymptotic (analytical) bands — fast and version-stable.
    Falls back to MC simulation if asymptotic bands are unavailable.
    """
    irfs = irf_obj.orth_irfs if orth else irf_obj.irfs
    point = irfs[:, response_idx, impulse_idx]

    try:
        stderr = irf_obj.stderr(orth=orth)
        z = abs(np.percentile(np.random.standard_normal(100_000), (signif / 2) * 100))
        se = stderr[:, response_idx, impulse_idx]
        lower, upper = point - z * se, point + z * se
    except Exception:
        # Fallback: Monte Carlo bootstrap (slower but robust)
        bands = irf_obj.err_band_mc(orth=orth, repl=500, seed=42)
        lower = bands.lower[:, response_idx, impulse_idx]
        upper = bands.upper[:, response_idx, impulse_idx]

    return point, lower, upper



df_var = prepare_macro_data(df_macro, df_tariffs, df_trade)

if not df_var.empty:
    df_var_diff = df_var.copy()
    for col in df_var.columns:
        is_stationary = check_stationarity(df_var[col])
        status = "stationary ✅" if is_stationary else "non-stationary → differencing 🔁"
        print(f"  {col}: {status}")
        if not is_stationary:
            df_var_diff[col] = df_var[col].diff()

    df_var_diff = df_var_diff.dropna()
    print(f"\n  Post-differencing shape: {df_var_diff.shape}")


    model = VAR(df_var_diff)

    k = df_var_diff.shape[1]
    max_safe_lags = max(1, (len(df_var_diff) // (k * k)) - 1)
    maxlags = min(12, max_safe_lags)

    lag_selection = model.select_order(maxlags=maxlags)
    optimal_lag = max(1, lag_selection.aic)   # floor at 1 — lag=0 is degenerate
    print(f"  ✅ Optimal lag via AIC: {optimal_lag} month(s)  (searched up to {maxlags})")

    var_fitted = model.fit(optimal_lag)

    print("\n" + "=" * 60)
    print("  VAR RESIDUAL CORRELATION MATRIX")
    print("=" * 60)
    print(var_fitted.resid_corr.round(4))
    print("=" * 60)

    irf = var_fitted.irf(12)
    cols = df_var_diff.columns.tolist()
    impulse_idx = cols.index('Cumulative_Tariff_Rate')

    response_vars   = ['CPI_Inflation', 'Global_Trade_Volume']
    response_labels = ['CPI Inflation', 'Global Trade Volume']
    response_colors = ['steelblue', 'darkorange']

    fig, axes = plt.subplots(1, 2, figsize=(14, 5))

    for ax, var, label, color in zip(axes, response_vars, response_labels, response_colors):
        response_idx = cols.index(var)
        point, lower, upper = _get_irf_bands(irf, impulse_idx, response_idx)
        periods = range(len(point))

        ax.plot(periods, point, color=color, linewidth=2.2, label='IRF')
        ax.fill_between(periods, lower, upper, alpha=0.20, color=color, label='95% CI')
        ax.axhline(0, color='black', linewidth=0.9, linestyle='--')
        ax.set_title(f'Tariff Shock → {label}', fontweight='bold', fontsize=12)
        ax.set_xlabel('Months After Shock')
        ax.set_ylabel('Response (first-differenced units)')
        ax.legend(fontsize=9)
        ax.grid(True, alpha=0.3)
        ax.set_xticks(list(periods))

    fig.suptitle(
        "Macroeconomic Impulse Response to a Tariff Shock (12-Month Horizon)",
        fontweight='bold', fontsize=14, y=1.02
    )
    plt.tight_layout()
    plt.show()

    print("\n📊 Forecast Error Variance Decomposition (FEVD) @ 12 months:")
    fevd = var_fitted.fevd(12)
    fevd_df = pd.DataFrame(
        {col: fevd.decomp[i][-1] for i, col in enumerate(cols)},
        index=cols
    ).T
    fevd_df.columns = [f'Explained by: {c}' for c in cols]
    print(fevd_df.round(4))

    # Optional: FEVD bar chart
    fig2, ax2 = plt.subplots(figsize=(10, 4))
    fevd_df.plot(kind='bar', ax=ax2, colormap='tab10', edgecolor='white')
    ax2.set_title('Forecast Error Variance Decomposition @ 12 Months', fontweight='bold')
    ax2.set_xlabel('Variable')
    ax2.set_ylabel('Proportion of Forecast Variance')
    ax2.set_xticklabels(cols, rotation=15, ha='right')
    ax2.legend(loc='upper right', fontsize=8)
    ax2.grid(axis='y', alpha=0.3)
    plt.tight_layout()
    plt.show()

else:
    print("\n⚠️  Insufficient overlapping data to execute VAR modelling.")
    print("    Check that df_macro, df_tariffs, and df_trade share a common date range.")
```

#### VAR

All variables were non-stationary and were differenced, which ensures the relationships are stable and not driven by trends.

The optimal lag length is 7 months based on AIC. This suggests that shocks in tariffs, inflation, and trade take several months to fully propagate through the system rather than occurring instantly.

Residual correlations are low, with absolute values below 0.2. Values close to zero indicate that most of the relationships between variables have been captured by the model.

**Takeaway:**  
The VAR is dynamically rich enough to capture delayed effects, uses a sufficient sample size for credibility, and shows no major signs of misspecification.

---

#### IRF (Impulse Response Functions)

For **CPI inflation**, the response is small, slightly negative at first, and returns toward zero within 6 to 8 months. The confidence intervals are wide and frequently include zero, which means the effect is not statistically strong. This implies that any inflationary impact from tariffs is weak, short-lived, and uncertain.

For **global trade volume**, the response is immediately negative and becomes substantially more negative over time, reaching its largest decline around month 7. The effect remains negative throughout the horizon, even as it partially recovers. This indicates a persistent and economically meaningful contraction in trade following a tariff shock.

**Takeaway:**  
The timing of responses matters. Trade reacts quickly and strongly, while inflation shows little consistent reaction and high uncertainty.

---

#### FEVD (12-Month Horizon)

Forecast error variance decomposition shows how much of each variable’s fluctuations are explained by shocks to itself versus other variables over a 12-month horizon. Higher values indicate stronger explanatory influence.

For the **cumulative tariff rate**, about 0.96 of its variation is explained by its own shocks. This means tariff changes are largely independent within the system and can be treated as an external policy driver.

For **CPI inflation**, about 0.87 of its variation is explained by its own shocks, while tariffs account for 0.09. This shows that inflation is mostly driven by its own internal dynamics, with tariffs playing a secondary role.

For **global trade volume**, about 0.59 of its variation is explained by tariff shocks, compared to 0.31 explained by its own shocks. This indicates that tariffs are the dominant driver of trade fluctuations over this horizon.

**Takeaway:**  
Tariffs act as an external force, strongly influencing trade but only modestly affecting inflation, while inflation remains largely self-driven.

## 13. Key Findings

1. **Trade wars caused significant but temporary market dislocations.**

- Event studies show no statistically significant directional shocks in the immediate trading window.
- Major tariff announcements also do not produce statistically significant shocks in broad equity indices (e.g., S&P 500). Markets largely pre-price or absorb these events.
- Extreme market stress is not caused by trade wars. Major anomalies align with systemic crises (e.g., 2020 pandemic), not tariff announcements.

2. **Volatility—not returns—is the primary transmission channel**

- Across equities and FX, the dominant impact is persistent volatility rather than directional price movement. Trade shocks reshape risk regimes instead of market levels.

- Markets are primarily internally driven. ~59% of market dynamics are explained by volatility, indicating endogenous factors dominate over tariff events.

3. **FX acts as a dynamic shock absorber—until it doesn’t**

- 2018–2024 (Trade War 1.0 & Recovery): Tariffs → CNY depreciation (offsetting tariff costs)

- 2025 onward (Trade War 2.0): Structural break → CNY appreciation

- This indicates limits to currency adjustment mechanisms under prolonged trade pressure.

4. **Sectoral divergence is the real impact**
   Trade-exposed sectors (e.g., semiconductors, autos) show higher returns and higher volatility, while domestic sectors remain more stable but with lower returns.

5. **No strong short-term causal damage to targeted sectors**
   DiD results show no statistically significant immediate impact, suggesting rapid market adjustment.

6. **Trade wars rewire—not reduce—global trade**
   Short-term decline in trade volume is followed by recovery via rerouting, with countries like Vietnam, Mexico, and India benefiting.

7. **Macroeconomic impact is asymmetric**
   Trade volume is persistently negatively affected, while inflation effects are weak and inconsistent. Trade wars act more as growth shocks than inflation drivers.

8. **Tariffs behave as exogenous policy shocks**
   VAR and FEVD results show tariffs strongly influence trade flows but are not driven by macroeconomic variables.

9. **Market impact is driven by tariff scope rather than intensity**
   High-intensity retaliatory tariffs show limited or mixed market effects, while broad, system-wide tariffs (large trade coverage) generate the most negative returns and highest volatility, indicating that scale and systemic reach matter more than tariff magnitude alone.

## 14. Strategic Reccomendations

#### For Policymakers

1. **Anticipate trade diversion**
   Protectionist policies reallocate trade rather than reduce it. Plan for third-country spillovers and supply chain shifts.

2. **Strengthen supply chain resilience**
   Invest in domestic capacity and strategic partnerships (friend-shoring) to mitigate persistent trade volume shocks.

3. **Monitor financial stability through volatility**
   Track FX volatility and sector-level dispersion as primary indicators of trade shock impact.

#### For Traders

1. **Trade volatility, not direction**
   Tariff events are better suited for options strategies and volatility-based approaches rather than directional bets.

2. **Rotate selectively into trade-exposed sectors**
   These sectors offer higher returns but with higher risk, requiring active management and hedging.

3. **Avoid overreacting to tariff headlines**
   Aggregate market responses are muted, so avoid panic-driven decisions.

## 15. Risks and Limitations

1. Data limitations
   Tariff data may not fully capture non-tariff barriers, and financial variables act as indirect proxies of economic impact.

2. Model assumptions
   GARCH may miss nonlinear regime shifts, while VAR differencing may remove long-term relationships.

3. Statistical insignificance in causal models
   DiD results may reflect either market efficiency or limited model sensitivity.

4. Concurrent COVID-19 and monetary policy effects may confound results
   Overlapping shocks during the sample period may blur the isolated impact of trade war policies.

5. Bilateral trade data needed for precise trade diversion measurement
   Aggregate trade data limits the ability to accurately capture rerouting effects across specific country pairs.
